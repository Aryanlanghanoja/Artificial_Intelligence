idf_vector = []
N = len(documents)

for word in vocab:
    df = sum([1 for doc in tokenized_docs if word in doc])
    idf = math.log(N / (1 + df))
    idf_vector.append(idf)

tf_idf_matrix = np.array(tf_vectors) * np.array(idf_vector)
tfidf_df = pd.DataFrame(tf_idf_matrix, columns=vocab)

print("\nTF-IDF Vectorization:")
